    **API client extensions** (`frontend/src/api/client.ts`):

    Add three new exported functions following the existing pattern (credentials: 'include'):

    ```typescript
    export interface SettingsResponse {
      configured: boolean
      last_sync_at: string | null
      last_sync_status: string | null
    }

    export async function getSettings(): Promise<SettingsResponse> {
      const res = await fetch('/api/settings', { credentials: 'include' })
      return res.json()
    }

    export async function saveSettings(accessUrl: string): Promise<{ok: boolean}> {
      const res = await fetch('/api/settings', {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        credentials: 'include',
        body: JSON.stringify({ access_url: accessUrl }),
      })
      return res.json()
    }

    export async function triggerSync(): Promise<{ok: boolean}> {
      const res = await fetch('/api/sync/now', {
        method: 'POST',
        credentials: 'include',
      })
      return res.json()
    }
    ```

      const seconds = Math.floor((Date.now() - new Date(dateStr).getTime()) / 1000)
      if (seconds < 60) return 'just now'
      if (seconds < 3600) return `${Math.floor(seconds / 60)}m ago`
